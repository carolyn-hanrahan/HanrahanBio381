x <- ggplot(data=d, aes(x = len, y = dose, fill = dose, group = dose)) +
  geom_density_ridges() +
  theme_ridges() + 
  theme(legend.position = "left")
  labs(title='Tooth Length by Vitamin C Treatment Group')
